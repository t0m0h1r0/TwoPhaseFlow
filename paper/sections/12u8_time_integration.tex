\subsubsection{U8：時間積分スイート（TVD-RK3 / AB2 / CN-ADI / Layer A,B,C）[Tier V]}
\label{sec:U8_time_integration}

\paragraph{検証対象．}
第~\ref{sec:time_int}章 で導入した時間積分の全要素を単体検証する：
TVD--RK3（§\ref{sec:time_int}：3 次 SSP），
AB2（§\ref{sec:time_int}：2 次 multistep），
CN--ADI（§\ref{sec:viscous_cn_defect}：粘性項 implicit ADI），
A/B/C 3 層粘性切替（§\ref{sec:viscous_3layer}：均一 / 中濃度差 / 強濃度差）．

\paragraph{設定．}
\begin{description}\setlength{\itemsep}{0pt}
  \item[U8-a] TVD--RK3 単体．スカラー ODE $\dot{u} = -\lambda u, \lambda=1, T=1$．
        $n_t \in \{4,8,\dots,512\}$．設計次数 3．
  \item[U8-b] AB2 単体．同 ODE．$n_t \in \{16,\dots,512\}$．設計次数 2．
  \item[U8-c] CN--ADI 拡散方程式．$\nu = 0.01$，1D 拡散方程式，
        $n_t \in \{10,\dots,320\}$．設計次数 2．
        \textbf{安定性}：CFL$_\nu \in \{0.5,1,5,10\}$ で CN は安定，
        FE は CFL$_\nu \ge 0.5$ で発散．
  \item[U8-d] A/B/C 3 層 CN--ADI（界面 viscosity ratio test）．
        $\mu_{\mathrm{ratio}} \in \{10, 100\}$，layer $\in \{$A, B, C$\}$，
        $n_t \in \{16,\dots,256\}$．Layer A=均一 / B=粘性勾配 / C=粘性 + 界面．
\end{description}

\paragraph{判定と次章への接続．}
\begin{itemize}\setlength{\itemsep}{0pt}
  \item U8-a TVD--RK3：slope $3.00$，設計 3 次完全達成．\checkmark
  \item U8-b AB2：slope $2.04$，設計 2 次完全達成．\checkmark
  \item U8-c CN--ADI：slope $2.00$，FE 比 安定性 $19$ 桁優位（CFL$_\nu=0.5$ にて）．
        §\ref{sec:viscous_cn_defect} の A--安定設計と整合．\checkmark
  \item U8-d Layer A/B/C：A 層は完全 2 次，B/C 層は ADI 分割で実効 1 次まで
        劣化．これは ADI--Strang 整合則の典型挙動であり，
        §\ref{sec:viscous_3layer} の予測通り．$\triangle$（既知の劣化を定量化）．
\end{itemize}
