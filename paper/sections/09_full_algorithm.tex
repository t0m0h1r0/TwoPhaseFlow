\section{アルゴリズムフローの厳密な擬似コード表現}
\label{sec:algorithm}

時刻 $t^n$ から $t^{n+1} = t^n + \Delta t$ への計算手順を，プログラミング実装を意識した厳密な擬似コード形式で記述する．

\paragraph{離散オペレータの定義}
まず，離散的な微分オペレータを以下のように定義する．
\begin{itemize}
    \item $\nabla_h \cdot \bm{q}$: ベクトル場 $\bm{q}$ の発散．本稿では，移流項の計算にはWENO5を，それ以外（圧力方程式など）では中心差分（Rhie-Chow補間を含む）を用いる．
    \item $\nabla_h s$: スカラー場 $s$ の勾配．CCD法または中心差分を用いる．
    \item $\mathcal{L}_h(\bm{u}) \equiv \nabla_h \cdot \left[ \frac{\mu}{Re} (\nabla_h \bm{u} + (\nabla_h \bm{u})^T) \right]$: 粘性項（ラプラシアン）オペレータ．本稿では内部の微分計算にCCD法を適用する．
\end{itemize}

\begin{tcolorbox}[algbox, title={時刻 $t^n \to t^{n+1}$ の厳密計算フロー}]

\textbf{BEGIN Time Step} ($t^n \to t^{n+1}$)

\begin{enumerate}
    \item \textbf{CLS Advection (TVD-RK3):}
    \begin{align*}
      \bm{F}^{(0)} &:= -\nabla_h \cdot (\psi^n \bm{u}^n) \quad (\text{WENO5}) \\
      \psi^{(1)} &:= \psi^n + \Delta t \, \bm{F}^{(0)} \\
      \bm{F}^{(1)} &:= -\nabla_h \cdot (\psi^{(1)} \bm{u}^{n+1}) \quad (\text{WENO5}) \\
      \psi^{(2)} &:= \frac{3}{4}\psi^n + \frac{1}{4}(\psi^{(1)} + \Delta t \, \bm{F}^{(1)}) \\
      \bm{F}^{(2)} &:= -\nabla_h \cdot (\psi^{(2)} \bm{u}^{n+1}) \quad (\text{WENO5}) \\
      \psi^* &:= \frac{1}{3}\psi^n + \frac{2}{3}(\psi^{(2)} + \Delta t \, \bm{F}^{(2)})
    \end{align*}

    \item \textbf{Reinitialization:} \\
    偏微分方程式 \eqref{eq:cls_reinit_iso} を仮想時間 $\tau$ について複数ステップ解き，$\psi^*$ を整形して $\psi^{n+1}$ を得る．
    \[ \psi^{n+1} \leftarrow \text{SolveReinitPDE}(\psi^*) \]

    \item \textbf{Properties Update:} \\
    FOR all grid points $(i,j)$ DO:
    \begin{align*}
        \phi^{n+1}_{i,j} &\leftarrow \text{Solve}[ H_\varepsilon(\phi) = \psi^{n+1}_{i,j} ] \quad (\text{Newton's method}) \\
        \rho^{n+1}_{i,j} &\leftarrow \rho(\phi^{n+1}_{i,j}) \\
        \mu^{n+1}_{i,j} &\leftarrow \mu(\phi^{n+1}_{i,j})
    \end{align*}
    END FOR

    \item \textbf{Curvature Calculation:} \\
    CCD法を用いて $\phi^{n+1}$ の各階微分 $D_x\phi, D_{xx}\phi, \dots$ を計算し，公式 \eqref{eq:curvature_2d} に代入して $\kappa^{n+1}$ を得る．

    \item \textbf{Predictor Step (Momentum Prediction):} \\
    予測速度 $\bm{u}^*$ を求める．粘性項をCrank-Nicolson法で扱うため，以下の線形方程式系を解く．
    \[ \left( \frac{\rho^{n+1}}{\Delta t} - \frac{1}{2}\mathcal{L}_h \right) \bm{u}^* = \left( \frac{\rho^{n+1}}{\Delta t} + \frac{1}{2}\mathcal{L}_h \right) \bm{u}^n - \rho^{n+1}\nabla_h \cdot (\bm{u}^n \bm{u}^n) + \bm{F}_{\text{ext}}^{n+1} \]
    ここで $\bm{F}_{\text{ext}}^{n+1}$ は重力項と表面張力項 $\frac{\kappa^{n+1}}{We}\nabla_h\psi^{n+1}$ の和である．
    この大規模な連立一次方程式を BiCGSTAB などの反復法ソルバーで解き，$\bm{u}^*$ を得る．

    \item \textbf{Pressure Poisson Equation:} \\
    次ステップの圧力 $p^{n+1}$ を求めるためのPoisson方程式を構築し，解く．
    \[ \nabla_h \cdot \left( \frac{1}{\rho^{n+1}} \nabla_h p^{n+1} \right) = \frac{1}{\Delta t} \nabla_h \cdot \bm{u}^* \]
    ここでオペレータ $\nabla_h, \nabla_h \cdot$ はRhie-Chow補間を内包する．
    この線形方程式系を反復法ソルバーで解き，$p^{n+1}$ を得る．

    \item \textbf{Corrector Step (Velocity Correction):} \\
    圧力勾配を用いて速度場を補正し，非圧縮性を満たす最終的な速度場 $\bm{u}^{n+1}$ を得る．
    \[ \bm{u}^{n+1} = \bm{u}^* - \frac{\Delta t}{\rho^{n+1}} \nabla_h p^{n+1} \]
\end{enumerate}

\textbf{END Time Step}

\end{tcolorbox}
