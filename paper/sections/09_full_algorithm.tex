\section{完全アルゴリズムフロー：離散形式}
\label{sec:algorithm}

本章では，第~\ref{sec:advection}章まで個別に導いた数値手法を統合し，
時刻 $t^n$ から $t^{n+1}$ への1タイムステップ全体を離散形式でまとめる．

\begin{enumerate}
  \item \textbf{§~\ref{sec:algo_operator_map}（演算子マッピング）：}
        NS 方程式の各項と離散ソルバーの対応関係を図示する．
  \item \textbf{§~\ref{sec:algo_flow}（全体フロー）：}
        7ステップ——CLS 移流・再初期化・物性更新・曲率・Predictor・PPE・Corrector——を
        離散形式で詳述する．
  \item \textbf{§~\ref{sec:algo_timestep}（タイムステップ制御）：}
        対流 CFL と毛管波制約から $\Delta t$ を決定する．
\end{enumerate}

\end{itemize}

\subsection{アルゴリズムの全体フロー}
\label{sec:algo_flow}
% ─────────────────────────────────────────────────────────────────────────────

時刻 $t^n$ から $t^{n+1} = t^n + \Delta t$ への計算手順を，離散化された形式で以下にまとめる．
添字 $(i,j)$ は格子点のインデックスを，上付き文字 $n, *, n+1$ は時間ステップを表す．
以下の離散化オペレータを用いる：
\begin{itemize}
  \item $D_x^{(1)}, D_y^{(1)}$：CCD 法による1次微分 $(\partial/\partial x,\, \partial/\partial y)$
  \item $D_x^{(2)}, D_y^{(2)}$：CCD 法による2次微分 $(\partial^2/\partial x^2,\, \partial^2/\partial y^2)$
  \item $D_{xy}^{(11)}$：CCD 法による混合微分 $\partial^2/\partial x\partial y$—
        $(D_{xy}^{(11)} f)_{i,j} = (D_y^{(1)}[D_x^{(1)} f])_{i,j}$（逐次2段スウィープ，$\mathcal{O}(h^6)$）
  \item $\mathcal{C}_{\text{CCD}}(\bu,\, q)$：速度 $\bu$ による CCD 中心差分移流オペレータ（\textbf{NS 対流項専用}）
          $\displaystyle= u\,D_x^{(1)} q + v\,D_y^{(1)} q$
          （\textbf{非保存形}；非圧縮条件 $\bnabla\cdot\bm{u}=0$ のもとで保存形と等価．
          \textbf{注意：} CLS 移流には Dissipative CCD による保存形演算子 $\mathrm{Div}_h^{\psi}$ を使用する；§~\ref{sec:advection_motivation} 参照）
  \item $\mathcal{L}_{\text{visc},x}(\mu,\, \bu)$：粘性応力テンソル $\bnabla\cdot\boldsymbol{\tau}$ の $x$ 成分\newline
          $\displaystyle= D_x^{(1)}\!\left[2\mu\,D_x^{(1)} u\right] + D_y^{(1)}\!\left[\mu\bigl(D_y^{(1)} u + D_x^{(1)} v\bigr)\right]$
  \item $\mathcal{L}_{\text{visc},y}(\mu,\, \bu)$：粘性応力テンソル $\bnabla\cdot\boldsymbol{\tau}$ の $y$ 成分\newline
          $\displaystyle= D_x^{(1)}\!\left[\mu\bigl(D_x^{(1)} v + D_y^{(1)} u\bigr)\right] + D_y^{(1)}\!\left[2\mu\,D_y^{(1)} v\right]$
  \item CLS 移流の発散 $\text{Div}_h^{\psi}(\psi\bm{u})$：Dissipative CCD フィルタ付き保存形フラックスの発散
          $\approx D_x^{(1)}(\psi u) + D_y^{(1)}(\psi v)$（保存形，詳細は§~\ref{sec:advection_motivation}）
  \item \textbf{PPE 演算子} $\mathcal{L}_{\mathrm{CCD}}^{\rho}$：CCD product-rule 展開（式~\eqref{eq:varrho_product_rule}，§~\ref{sec:ccd_2d_full}）による変密度ラプラシアン
          $\displaystyle= \frac{1}{\rho}(D_x^{(2)}p + D_y^{(2)}p)
            - \frac{D_x^{(1)}\rho}{\rho^2}D_x^{(1)}p
            - \frac{D_y^{(1)}\rho}{\rho^2}D_y^{(1)}p$（$\mathcal{O}(h^6)$）
\end{itemize}

\textbf{1. CLS 移流 (Advection)} \\
保存形レベルセット変数 $\psi^n$ を，速度場 $\bm{u}^n$ を用いて時間発展させる．
ここでは TVD-RK3（3段3次）を用いる．
\begin{align*}
  \psi^{(1)}_{i,j} &= \psi^n_{i,j} - \Delta t \, \text{Div}_h^{\psi}(\psi^n \bm{u}^n)_{i,j} \\
  \psi^{(2)}_{i,j} &= \frac{3}{4}\psi^n_{i,j} + \frac{1}{4}\psi^{(1)}_{i,j} - \frac{1}{4}\Delta t \, \text{Div}_h^{\psi}(\psi^{(1)} \bm{u}^n)_{i,j} \\
  \psi^*_{i,j} &= \frac{1}{3}\psi^n_{i,j} + \frac{2}{3}\psi^{(2)}_{i,j} - \frac{2}{3}\Delta t \, \text{Div}_h^{\psi}(\psi^{(2)} \bm{u}^n)_{i,j}
\end{align*}
ここで $\text{Div}_h^{\psi}(\cdot)$ は Dissipative CCD フィルタ付き保存形発散オペレータである（§~\ref{sec:advection_motivation} 参照）．
\textbf{各ステージ完了後に} $\psi \leftarrow \max(0,\min(1,\psi))$ のクランプを適用すること
（Dissipative CCD は TVD 性を持たないため必須；注意~\ref{warn:adv_clamp}）．

\textbf{2. 再初期化 (Reinitialization)} \\
移流によって鈍化した界面プロファイル $\psi^*$ を整形する．
式~\eqref{eq:cls_reinit_iso} で表される偏微分方程式を，仮想時間 $\tau$ について固定ステップ数解く：
\[ \psi^{n+1} = \text{Reinitialize}(\psi^*,\; n_{\mathrm{reinit}}) \]
\textbf{仮想時間刻み：}
\begin{equation}
  \Delta\tau = \min\!\left(
    \underbrace{\frac{0.5\,h^2}{2\,N_{\mathrm{dim}}\,\varepsilon}}_{\text{放物安定条件}},\;
    \underbrace{0.5\,h}_{\text{双曲 CFL}}
  \right)
\end{equation}
（式~\eqref{eq:dtau_reinit_def}，§~\ref{sec:cls_compression} より再掲）
ここで $h = \min(\Delta x, \Delta y)$，$N_{\mathrm{dim}}$ は空間次元数（2D では $N_{\mathrm{dim}}=2$），
$\varepsilon$ は界面幅パラメータ（§~\ref{sec:levelset}）．
放物安定条件は拡散項 $\varepsilon\Delta\psi$ の陽的処理に対する CFL 制約であり，
TVD-RK3 の安全率 0.5 を含む（$\varepsilon \sim h$ では放物条件が支配的）．

\textbf{ステップ数：} デフォルト $n_{\mathrm{reinit}} = 4$（\texttt{config.numerics.reinit\_steps} で変更可）．
収束判定による停止は行わず，固定ステップ数を毎タイムステップ実行する
（$n_{\mathrm{reinit}} = 4$ は $\Delta\tau \sim 0.5h$ のとき仮想時間 $\tau_{\mathrm{total}} \approx 2h$ に相当し，
界面プロファイルの整形に十分な仮想時間を確保する；§~\ref{sec:cls_compression} のアルゴリズム~\ref{alg:cls_reinit_dccd} 参照）．

\textbf{3. グリッド・物性更新 (Update Properties)} \\
更新された $\psi^{n+1}_{i,j}$ に基づき，各格子点での距離関数 $\phi$ および物理的物性値を再計算する．
物性補間 $\rho,\,\mu$ には $\psi$ を使い，幾何量（法線・曲率）には $\phi$ を使う（§~\ref{sec:notation} 参照；\textbf{$\rho(\phi)$，$\mu(\phi)$ と書いてはならない}）．
\begin{align*}
  \phi^{n+1}_{i,j} &= \varepsilon\ln\!\left(\frac{\psi^{n+1}_{i,j}}{1-\psi^{n+1}_{i,j}}\right)
                    \quad (\text{ロジット関数，§~\ref{sec:newton_inversion}；飽和域のみニュートン法}) \\
  \rho^{n+1}_{i,j} &= \rho_l + (\rho_g-\rho_l)\,\psi^{n+1}_{i,j} \\
  \mu^{n+1}_{i,j}  &= \mu_l + (\mu_g-\mu_l)\,\psi^{n+1}_{i,j}
\end{align*}
CLS 法では $\psi$ 自体が平滑化 Heaviside 関数であるため（§~\ref{sec:why_cls}），線形補間 $\rho(\psi) = \rho_l + (\rho_g - \rho_l)\psi$ は $\rho(\phi) = \rho_l + (\rho_g - \rho_l)H_\varepsilon(\phi)$ と等価である．

\noindent\textbf{注（平滑化デルタ関数の自動更新）：}
CLS 法では平滑化デルタ関数 $\delta_\varepsilon \approx |\nabla\psi|$ を独立フィールドとして保持しない．
CSF 体積力（Step~5，式~\eqref{eq:balanced_surface} の $\nabla\psi^{n+1}$）は
常に更新済み $\psi^{n+1}$ から CCD 微分を直接評価するため，
再初期化後の整合性は自動的に保たれる（明示的な $\delta_\varepsilon$ の再計算ステップは不要）．

\noindent\textbf{注（時刻統一）：}
$\rho^{n+1}$，$\mu^{n+1}$ は Step 4–7 で終始同一の値を使用する
（中間更新なし；変密度 IPC の正確性に必須）．

\medskip
\textbf{4. 曲率計算 (Curvature)} \\
$\phi^{n+1}$ の空間微分をCCD法で高精度に計算し，式~\eqref{eq:curvature_2d} に直接代入して曲率を求める．
\begin{align*}
 (\phi_x)_{i,j} &= D_x^{(1)} \phi^{n+1}_{i,j}, \quad
 (\phi_y)_{i,j} = D_y^{(1)} \phi^{n+1}_{i,j} \\
 (\phi_{xx})_{i,j} &= D_x^{(2)} \phi^{n+1}_{i,j}, \quad
 (\phi_{yy})_{i,j} = D_y^{(2)} \phi^{n+1}_{i,j}, \quad
 (\phi_{xy})_{i,j} = D_{xy}^{(11)} \phi^{n+1}_{i,j}
\end{align*}
\begin{equation*}
 \kappa^{n+1}_{i,j} = -\frac{(\phi_y)_{i,j}^2 (\phi_{xx})_{i,j}
                              - 2(\phi_x)_{i,j}(\phi_y)_{i,j}(\phi_{xy})_{i,j}
                              + (\phi_x)_{i,j}^2 (\phi_{yy})_{i,j}}
                           {\max\!\bigl((\phi_x)_{i,j}^2 + (\phi_y)_{i,j}^2,\;
                            \varepsilon_{\text{norm}}^2\bigr)^{3/2}}
\end{equation*}
ここで $\varepsilon_{\text{norm}}$ は勾配ゼロ点の正則化パラメータ（式~\eqref{eq:curvature_clamped} で定義）．

\textbf{5. 運動量予測 (Predictor Step, AB2 + IPC)} \\
対流項に AB2 外挿（係数 $3/2$，$-1/2$），粘性項に CN，IPC により前時刻圧力 $-\bnabla p^n$ を陽的に加える
（式~\eqref{eq:predictor_ab2_ipc}）．

\noindent\textbf{スタートアップ：}$n=0$ のみ対流項を前進 Euler で代用し，
$n \ge 1$ で AB2 に移行する（寄生根減衰により全体精度 $\Ord{\Delta t^2}$ 保持；
§~\ref{sec:ab2_ipc}，注意~\ref{warn:ab2_startup}）．

\medskip
\underline{$x$成分（$u^*_{i,j}$ を求める，$n \ge 1$）：}
\begin{align*}
  \frac{\rho^{n+1}_{i,j}}{\Delta t}\bigl(u^*_{i,j} - u^n_{i,j}\bigr)
  &= -\rho^{n+1}_{i,j}\Bigl[
      \tfrac{3}{2}\!\bigl(u^n_{i,j}\,D_x^{(1)} u^n_{i,j} + v^n_{i,j}\,D_y^{(1)} u^n_{i,j}\bigr)
      -\tfrac{1}{2}\!\bigl(u^{n-1}_{i,j}\,D_x^{(1)} u^{n-1}_{i,j} + v^{n-1}_{i,j}\,D_y^{(1)} u^{n-1}_{i,j}\bigr)
    \Bigr] \\
  &\quad + \frac{1}{2}\Bigl[D_x^{(1)}\!\bigl(2\mu^{n+1}_{i,j} D_x^{(1)} u^*\bigr)
                          + D_y^{(1)}\!\bigl(\mu^{n+1}_{i,j}(D_y^{(1)} u^* + D_x^{(1)} v^n)\bigr) \\
  &\qquad\qquad    + D_x^{(1)}\!\bigl(2\mu^{n+1}_{i,j} D_x^{(1)} u^n\bigr)
                          + D_y^{(1)}\!\bigl(\mu^{n+1}_{i,j}(D_y^{(1)} u^n + D_x^{(1)} v^n)\bigr)\Bigr]_{i,j} \\
  &\quad - \bigl(D_x^{(1)} p^n\bigr)_{i,j}
         + \frac{1}{We}\,\kappa^{n+1}_{i,j}\,\bigl(D_x^{(1)}\psi^{n+1}\bigr)_{i,j}
\end{align*}
\textbf{注：} CSF 体積力 $(\kappa/We)\,D_x^{(1)}\psi$ は Predictor の $R^{\text{explicit}}_x$ に含まれる
（Balanced-Force 条件，式~\eqref{eq:balanced_pressure}，\eqref{eq:balanced_surface}）．

\underline{$y$成分（$v^*_{i,j}$ を求める，$n \ge 1$）：}
\begin{align*}
  \frac{\rho^{n+1}_{i,j}}{\Delta t}\bigl(v^*_{i,j} - v^n_{i,j}\bigr)
  &= -\rho^{n+1}_{i,j}\Bigl[
      \frac{3}{2}\!\bigl(u^n_{i,j}\,D_x^{(1)} v^n_{i,j} + v^n_{i,j}\,D_y^{(1)} v^n_{i,j}\bigr)
      -\frac{1}{2}\!\bigl(u^{n-1}_{i,j}\,D_x^{(1)} v^{n-1}_{i,j} + v^{n-1}_{i,j}\,D_y^{(1)} v^{n-1}_{i,j}\bigr)
    \Bigr] \\
  &\quad + \frac{1}{2}\Bigl[D_x^{(1)}\!\bigl(\mu^{n+1}_{i,j}(D_x^{(1)} v^n + D_y^{(1)} u^n)\bigr)
                          + D_y^{(1)}\!\bigl(2\mu^{n+1}_{i,j} D_y^{(1)} v^*\bigr) \\
  &\qquad\qquad    + D_x^{(1)}\!\bigl(\mu^{n+1}_{i,j}(D_x^{(1)} v^n + D_y^{(1)} u^n)\bigr)
                          + D_y^{(1)}\!\bigl(2\mu^{n+1}_{i,j} D_y^{(1)} v^n\bigr)\Bigr]_{i,j} \\
  &\quad - \bigl(D_y^{(1)} p^n\bigr)_{i,j}
         - \frac{\rho^{n+1}_{i,j}}{Fr^2}
         + \frac{1}{We}\,\kappa^{n+1}_{i,j}\,\bigl(D_y^{(1)}\psi^{n+1}\bigr)_{i,j}
\end{align*}
\noindent\textbf{注（CN 離散化と ADI 実装の関係）：}
上式は形式的な CN 離散化を示す．実装では ADI 分裂により，クロス微分項（$D_y^{(1)}(\mu D_x^{(1)} v)$ 等）は $n$ 時刻の値で陽的に評価し，直接微分項（$D_x^{(1)}(2\mu D_x^{(1)} u^*)$ 等）のみを陰的に解く．

Step 5 完了後：対流項 $\mathcal{C}^n_{i,j}$（$x$ 成分：$u^n D_x^{(1)}u^n + v^n D_y^{(1)}u^n$，$y$ 成分同様）
をバッファに保存し（AB2 の $n-1$ 項として次ステップで利用；追加メモリ：$2N_xN_y$ 実数値），
速度場 $\bu^n$ は上書きして次の Corrector で $\bu^{n+1}$ に進める．
上式は CN の「解きたい方程式」であり，$D_y^{(1)} u^*$ 等のクロス項も形式上は陰的側に現れる．
実際の ADI スウィープでは各方向の対角項のみを陰的に解き，クロス項は $\bu^n$ で陽的に評価する
（$R^{\text{explicit}}_x$ に含まれる）．
ADI の陰的係数には完全な CCD 6次演算子（§~\ref{sec:time_int} の $\mathcal{L}_{\text{visc},x}$）の代わりに
2次中心差分の分割係数 $\tilde\lambda^\pm \equiv \mu^\pm/\Delta x^2$ を用いる．

\noindent\textbf{ADI 陰的係数に2次 FD を用いる理由：}
CCD のブロック連成構造は ADI Thomas 法と直接互換でないため，
陰的スウィープには2次 FD を使用する
（対角成分 $\Ord{h^2}$，クロス粘性項は CCD $\Ord{h^6}$ で陽的評価）．
$Re \gg 1$ では粘性項 $\Ord{h^2/Re}$ は CSF モデル誤差と同次以下に収まる．
ADI 分割誤差 $\Ord{\Delta t^2}$ は IPC と同次で時間精度に影響しない
（詳細：§~\ref{sec:pressure_accuracy_summary}）．

\medskip
$x$ スウィープ（$j$ 行固定，$\tilde\lambda^\pm_{i,j} \equiv \mu^\pm_{i,j}/\Delta x^2$）：
\begin{equation*}
\begin{split}
  &-\tilde\lambda^-_{i,j}\,u^*_{i-1,j}
  + \!\left(\frac{\rho_{i,j}}{\Delta t} + \tilde\lambda^-_{i,j} + \tilde\lambda^+_{i,j}\right)\!u^*_{i,j}
  - \tilde\lambda^+_{i,j}\,u^*_{i+1,j} \\
  &\quad= \left(\frac{\rho_{i,j}}{\Delta t} - \tilde\lambda^-_{i,j} - \tilde\lambda^+_{i,j}\right)u^n_{i,j}
  + \tilde\lambda^-_{i,j} u^n_{i-1,j}
  + \tilde\lambda^+_{i,j} u^n_{i+1,j}
  + (R^{\text{explicit}}_x)_{i,j}
\end{split}
\end{equation*}
$(R^{\text{explicit}}_x)_{i,j}$ は対流項・クロス粘性・CSF 表面張力の陽的寄与（§~\ref{sec:advection} 参照；CSF 体積力 $(\kappa/We)D_x^{(1)}\psi$ を含む）．
$y$ スイープ（$i$ 列固定）は $x \leftrightarrow y$，$\tilde\lambda \to \hat\lambda \equiv \mu/\Delta y^2$ で対称的に構成する．
2段 ADI の分割誤差は $\Ord{\Delta t^2}$ であり，IPC と同次のため全体精度に影響しない．

\textbf{5 の後処理．HFE（圧力場延長）——分相 PPE 解法でのみ使用} \\
\noindent\textbf{現行の CSF + smoothed Heaviside 一括解法ではこのステップは省略する．}
圧力場 $p$ が $\varepsilon$-regularized で既に滑らかなため，
HFE は不要であり，むしろ有害である（§~\ref{sec:ppe_discretization_choice}）．

将来の分相 PPE 解法（§~\ref{sec:split_ppe_motivation}）では，
圧力場が Young--Laplace ジャンプ $[p]_\Gamma = \sigma\kappa$ を陽に持つため，
PPE 求解前に $p^n$ を HFE で界面越しに延長する：
\[
  p^n_{\mathrm{ext}} = \text{HermiteExtension}(p^n,\; \phi^{n+1},\; \hat{\bm{n}}^{n+1})
\]
これにより CCD ステンシルが界面をまたいで参照する際の Gibbs 振動を除去し，
$\bnabla p^n$ の $\Ord{h^6}$ 精度を保つ（§~\ref{sec:field_extension}）．

\textbf{6. 圧力 Poisson 方程式の求解（IPC 増分法）(Pressure Solve)} \\
IPC 法では，絶対圧力 $p^{n+1}$ ではなく\textbf{圧力増分 $\delta p \equiv p^{n+1} - p^n$} を求解する．
CCD 演算子は標準版と\textbf{完全に同一}であり，未知変数名が変わるだけである
（離散 PPE，CCD product-rule 展開，§~\ref{sec:ccd_2d_full}，$\mathcal{O}(h^6)$）：
\begin{equation}
  \left[\frac{D_x^{(2)}(\delta p)}{\rho^{n+1}}
  - \frac{D_x^{(1)}\rho^{n+1}}{(\rho^{n+1})^2}D_x^{(1)}(\delta p)\right]_{i,j}
  + \left[\frac{D_y^{(2)}(\delta p)}{\rho^{n+1}}
  - \frac{D_y^{(1)}\rho^{n+1}}{(\rho^{n+1})^2}D_y^{(1)}(\delta p)\right]_{i,j}
  = \frac{1}{\Delta t}\Bigl(D_x^{(1)}\tilde{u}^*_{i,j} + D_y^{(1)}\tilde{v}^*_{i,j}\Bigr)
  \label{eq:PPE_CCD_fullalgo}
\end{equation}
右辺は DCCD フィルタ後速度（式~\eqref{eq:dccd_filter_physical}，$\varepsilon_d=1/4$）の
CCD 発散（式~\eqref{eq:dccd_ppe_rhs}）のみから構成される．
現行ソルバでは HFE は省略される（§~\ref{sec:ppe_discretization_choice}）．
分相 PPE 解法を使用する場合は Step 5 の HFE により圧力場が延長済みとなるため，
右辺への追加補正項は不要である．
\textbf{境界条件：} $\partial(\delta p)/\partial n = 0$（ノイマン条件）—
$\partial p^{n+1}/\partial n = 0$ かつ $\partial p^n/\partial n = 0$ より，
$\partial(\delta p)/\partial n = \partial(p^{n+1}-p^n)/\partial n = 0$ が成立する．
詳細な反復アルゴリズム（Step 6.0–6.5）は下の algbox 参照．

\textbf{7. 速度・圧力補正 (Corrector Step, IPC)} \\
圧力増分 $\delta p$ を用いて速度と圧力を同時に更新する：
\begin{align*}
u^{n+1}_{i,j} &= u^*_{i,j}
  - \frac{\Delta t}{\rho^{n+1}_{i,j}}\,\bigl(D_x^{(1)} \delta p\bigr)_{i,j} \\
v^{n+1}_{i,j} &= v^*_{i,j}
  - \frac{\Delta t}{\rho^{n+1}_{i,j}}\,\bigl(D_y^{(1)} \delta p\bigr)_{i,j} \\
p^{n+1}_{i,j} &= p^n_{i,j} + \delta p_{i,j}
\end{align*}
この $\bm{u}^{n+1}$ は CCD 発散の意味で非圧縮条件を満たす：
$D_x^{(1)} u^{n+1} + D_y^{(1)} v^{n+1} \approx 0$
（DCCD フィルタが $2\Delta x$ モードを除去；Rhie--Chow Balanced-Force 補正は Predictor 段で適用済み；§~\ref{sec:dccd_decoupling}，§~\ref{sec:rc_balanced_force}）．
$p^{n+1}$ は次ステップの IPC Predictor で $p^n$ として使用される．

\subsubsection{CCD-Poisson ソルバーの詳細（Step 6 展開版）}
\label{sec:algo_ppe_detail}

\textbf{入力：} 予測速度 $\bu^*$，密度場 $\rho^{n+1}$，前時刻圧力 $p^n$（IPC 増分との整合に使用），曲率場 $\kappa^{n+1}$（CSF 表面張力に使用），
仮想時間刻み $\Delta\tau$，収束許容誤差 $\varepsilon_{\mathrm{tol}}$

\textbf{求解対象：} 圧力増分 $\delta p \equiv p^{n+1} - p^n$（初期反復値 $\delta p^0 = 0$）

\medskip
\textbf{前処理：}
\begin{enumerate}
  \item[(6.0)] \textbf{密度勾配の事前計算：}
  各行・列に CCD を適用し，$(D_x^{(1)}\rho)_{i,j}$ および $(D_y^{(1)}\rho)_{i,j}$ を計算（物性更新後1回）．
\end{enumerate}

\textbf{右辺の構築：}
\begin{enumerate}
  \item[(6.1)] DCCD フィルタ後 CCD 発散を計算（式~\eqref{eq:dccd_ppe_rhs}）：
  \[
    q_{i,j} = \frac{1}{\Delta t}\Bigl(D_x^{(1)}\tilde{u}^*_{i,j} + D_y^{(1)}\tilde{v}^*_{i,j}\Bigr)
  \]
  ここで $\tilde{u}^*, \tilde{v}^*$ は $\varepsilon_d=1/4$ フィルタ後速度（式~\eqref{eq:dccd_filter_physical}）．
  現行ソルバでは smoothed Heaviside により圧力場が既に滑らかなため，追加補正は不要．
  分相 PPE 解法使用時は HFE（Step 5 後処理）により圧力場を延長済みとする．
\end{enumerate}

\textbf{仮想時間反復ループ（$m = 0, 1, 2, \dots$）：}
\begin{enumerate}
  \item[(6.2)] \textbf{CCD 演算子の matrix-free 評価} $\mathcal{L}_{\mathrm{CCD}}^{\rho}\delta p^m$
  （$\delta p^0 = 0$）：
  \begin{enumerate}
    \item $x$ 方向 CCD スウィープ（各行 $j$）：$(D_x^{(1)}\delta p^m)_{i,j}$，$(D_x^{(2)}\delta p^m)_{i,j}$ を計算
    \item $y$ 方向 CCD スウィープ（各列 $i$）：$(D_y^{(1)}\delta p^m)_{i,j}$，$(D_y^{(2)}\delta p^m)_{i,j}$ を計算
    \item 演算子値の組み立て（式~\eqref{eq:L_CCD_2d_full}）：
    \begin{align}
      \bigl(\mathcal{L}_{\mathrm{CCD}}^{\rho}\delta p^m\bigr)_{i,j}
      &= \frac{(D_x^{(2)}\delta p^m)_{i,j} + (D_y^{(2)}\delta p^m)_{i,j}}{\rho_{i,j}}
      \notag\\
      &\quad
      - \frac{(D_x^{(1)}\rho)_{i,j}}{\rho_{i,j}^2}(D_x^{(1)}\delta p^m)_{i,j}
      - \frac{(D_y^{(1)}\rho)_{i,j}}{\rho_{i,j}^2}(D_y^{(1)}\delta p^m)_{i,j}
    \end{align}
  \end{enumerate}

  \item[(6.3)] \textbf{残差の計算と収束判定：}
  \[
    \varepsilon^m = \bigl\|\mathcal{L}_{\mathrm{CCD}}^{\rho}\delta p^m - q\bigr\|_2
  \]
  $\varepsilon^m < \varepsilon_{\mathrm{tol}}$ ならば収束として反復終了 $\rightarrow$ Step 6.5

  \item[(6.4)] \textbf{補正の求解（欠陥補正 Step 2，低次演算子 $L_L$ による方向分離スウィープ）：}

  欠陥 $d^m = q - \mathcal{L}_{\mathrm{CCD}}^{\rho}\delta p^m$（Step 6.2--6.3 で算出）に対し，
  低次演算子 $L_L = L_{L,x} + L_{L,y}$（2次 FD 三重対角，式~\eqref{eq:ll_split}）で
  補正量を求解する（§~\ref{sec:dc_sweep}）：

  \medskip
  Step A（$x$ 方向 Thomas 法，各行 $j$ 独立）：
  \[
    \bigl[\tfrac{1}{\Delta\tau}\,\mathbf{I} - L_{L,x}\bigr]\,\tilde{p}^{\,\cdot,j}
    = d^{m,\cdot,j}
    \quad \forall\, j = 0,\dots,N_y-1
  \]
  Neumann 境界は §~\ref{sec:ccd_neumann_bc} の手順で $\tilde{p}'_0 = 0$ を強制代入．

  Step B（$y$ 方向 Thomas 法，各列 $i$ 独立）：
  \[
    \bigl[\tfrac{1}{\Delta\tau}\,\mathbf{I} - L_{L,y}\bigr]\,\delta p_{\mathrm{corr}}^{i,\cdot}
    = \tilde{p}^{\,i,\cdot}
    \quad \forall\, i = 0,\dots,N_x-1
  \]

  \textbf{解の更新（欠陥補正 Step 3）：}
  $\delta p^{m+1} = \delta p^m + \omega\,\delta p_{\mathrm{corr}}$
  （$\omega = \Delta\tau$；§~\ref{sec:dc_terminology}）．

  \medskip
  \noindent 注：Krylov 型実装（分割誤差なし）では GMRES/CG 法で
  $\mathcal{L}_{\mathrm{CCD}}^{\rho}\delta p = q$ を直接解く；
  matrix-free 積ベクトルに (6.2) の CCD スウィープを利用する
  （注意~\ref{warn:ppe_splitting} Approach 2）．

  $m \leftarrow m+1$ として (6.2) へ戻る
\end{enumerate}

\textbf{収束後：}
\begin{enumerate}
  \item[(6.5)] $\delta p = \delta p^m$（収束した増分を採用）；$p^{n+1} = p^n + \delta p$（Step 7 へ引き渡し）
\end{enumerate}

\textbf{出力：} $\delta p$（$\Ord{h^6}$ 精度の圧力増分），$p^{n+1} = p^n + \delta p$，総反復回数

\subsection{タイムステップ制御}
\label{sec:algo_timestep}
% ─────────────────────────────────────────────────────────────────────────────

各タイムステップの開始時に，以下の2制約のうち最も厳しい条件で $\Delta t$ を決定する
（詳細な導出は §~\ref{sec:stability} 参照）：

\begin{equation}
  \Delta t = C_{\mathrm{CFL}} \cdot \min\!\left(
    \underbrace{\left[\max\!\left(\frac{|u|_{\max}}{\Delta x}+\frac{|v|_{\max}}{\Delta y}\right)\right]^{-1}}_{\Delta t_{\mathrm{adv}}\text{：対流 CFL（式~\eqref{eq:dt_adv}）}},\;
    \underbrace{\sqrt{\frac{(\rho_l+\rho_g)\,h_{\min}^3}{2\pi\sigma}}}_{\Delta t_\sigma\text{：毛管波制約（式~\eqref{eq:dt_sigma}）}}
  \right)
  \label{eq:dt_full_algo}
\end{equation}

ここで $C_{\mathrm{CFL}} \le 0.5$ は安全率（デフォルト 0.45），$h_{\min} = \min(\Delta x, \Delta y)$ である．
粘性項は Crank--Nicolson 法（無条件安定）で処理するため，等粘性系では粘性 CFL 制約は不要である．
ただし大密度比（$\mu_l/\mu_g \gg 1$）の気液系では，クロス微分項 $\partial_y(\mu\partial_x v)$ の
陽的処理に起因する実質的な粘性制約が界面近傍で復活する場合がある（§~\ref{sec:stability} 参照）．
