\subsubsection{U6：Split-PPE + DC + HFE [Tier IV]}
\label{sec:U6_split_ppe_dc_hfe}

\paragraph{検証対象．}
第~\ref{sec:ppe_solve}章 で導入した split--PPE 圧力系
（§\ref{sec:split_ppe_framework}：密度比 $\rho_l/\rho_g$ 大の分離方程式構造），
defect--correction（DC；§\ref{sec:dc_pure_fccd_convergence}）の
$\omega$--収束領域，および Hermite Field Extension
（HFE；§\ref{sec:hfe}：界面外挿による $\rho^{-1}\nabla p$ の非対称化抑制）の
\textbf{単体精度}を確認する．

\paragraph{単体テスト．}
\begin{description}\setlength{\itemsep}{0pt}
  \item[U6-a] DC $\omega$--収束図．$N=32$ 円界面 $R=0.25$，
        $\rho_l \in \{1, 100, 1000\}$，relaxation $\omega \in \{0.2,0.3,0.5,0.7,0.9\}$，
        $k_{\mathrm{DC}} \in \{2,3\}$．stall 判定 + 残差ヒストリ．
  \item[U6-b] $N$ 細分化 / $\rho$ 比依存性．$\rho_l = 1000$, $k_{\mathrm{DC}}=3$，
        $N \in \{32,64,128\}$．密度比 $\rho_l \in \{1,10,100,1000\}$ at $N=64$．
  \item[U6-c] HFE 単独：1D MMS（外挿ステンシル単体精度），
        2D MMS（$\rho^{-1}\nabla p$ の界面跨ぎ精度の \textbf{max} と \textbf{med}）．
\end{description}

\paragraph{評価．}
\begin{itemize}\setlength{\itemsep}{0pt}
  \item U6-a DC $\omega$--収束：$\rho_l = 1$ は安定収束；
        $\rho_l \ge 100$ は $\omega \ge 0.5$ で stall．
        $k_{\mathrm{DC}}=3$ は $k=2$ から残差を $\sim 25\%$ 削減．
        §\ref{sec:dc_pure_fccd_convergence} の理論域と一致．\checkmark
  \item U6-b $N/\rho$ sweep：高密度比 $\rho_l=1000$ では
        slope $\approx 0.78$（$N=32 \to 128$）に律速．これは
        \textbf{split--PPE の構造的制約}であり，§\ref{sec:varrho_ppe_limitation} で
        導入する $\varrho$--PPE / lumped--PPE 切替の動機となる．
        $\rho_l \le 10$ は $\Ord{h^2}$ 通常収束．\checkmark（限界明示）．
  \item U6-c HFE：1D 設計 6 次完全達成；2D は med で 3 次以上，
        max で 5 次（界面跨ぎセル 1 セル分の劣化が支配）．
        §\ref{sec:hfe} 設計仕様と整合．\checkmark
\end{itemize}
