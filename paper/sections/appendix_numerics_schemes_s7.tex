\section{CCD 離散化ポアソン問題に対するソルバー比較}
\label{app:ccd_solver_benchmark}

\subsection{主張}

\begin{center}
\textbf{%
  1 次元ポアソン問題において，CCD 離散化を用いることで%
  FD2 より 3--8 桁高い精度を達成でき，%
  CCD 系の反復ソルバーの中では BiCGSTAB が%
  行列非保持の仮想時間法より $O(N)$ 対 $O(N^2)$ の%
  反復回数優位を持つ．}
\end{center}

\subsection{問題設定と比較対象}

1 次元ポアソン問題
\begin{equation}
  \frac{\mathrm{d}^2\phi}{\mathrm{d}x^2} = S(x),
  \qquad x \in [0,1],\quad \phi(0)=\phi(1)=0,
  \label{eq:bench_poisson}
\end{equation}
厳密解 $\phi = \sin(2\pi x)$，$S = -(2\pi)^2\sin(2\pi x)$
を対象として，以下の 4 つの手法を比較する．

\begin{enumerate}
  \item \textbf{FD2 $+$ SOR}：標準 3 点中心差分（$O(h^2)$）$+$ 最適 SOR
        （緩和係数 $\omega_{\mathrm{opt}} = 2/(1+\sin(\pi h))$，古典的基準手法）
  \item \textbf{CCD $+$ Direct LU}：6 次精度 CCD $+$ 密行列 LU 分解（1 ステップ）
  \item \textbf{CCD $+$ BiCGSTAB}：6 次精度 CCD 係数行列 $+$ Krylov 反復
  \item \textbf{CCD $+$ 仮想時間}：$\partial\phi/\partial\tau = \phi'' - S$，
        前進 Euler，$\Delta\tau = 0.9 \times 2/\rho(D_2)$，行列非保持
\end{enumerate}

収束判定は $\|\mathrm{res}\|_\infty < 10^{-8}$（Direct LU は機械イプシロン相当）．
仮想時間法は $O(N^2)$ 反復スケーリングのため $N \leq 64$ のみ実測し，
$N \geq 128$ は $O(N^2)$ 外挿で推定する．

\subsection{理論的背景}

\paragraph{FD2 $+$ SOR の収束．}
最適 SOR スペクトル半径は $\rho_{\mathrm{SOR}} = \omega_{\mathrm{opt}} - 1 \approx 1 - 2\pi/N$，
収束ステップ数は $O(N)$：
\begin{equation}
  k_{\mathrm{SOR}} \approx \frac{\ln(|\Delta\phi_0|/\varepsilon)}{2\pi/N}
  = O(N).
\end{equation}

\paragraph{CCD $+$ 仮想時間の収束．}
仮想時間の収束率は $\phi'' - S$ を支配する固有モードの周波数に依存する．
初期誤差が $\phi_0 = 0$（ゆえに誤差 $= \sin(2\pi x)$，すなわちモード $k=2$）の場合：
\begin{equation}
  k_{\mathrm{PT}}
  \approx \frac{\ln(\|r_0\|/\varepsilon)}{|{\Delta\tau \cdot \lambda_2}|}
  \approx \frac{\ln(\|r_0\|/\varepsilon)}{0.9 \times 2(2\pi)^2/\rho(D_2)}
  = O(N^2),
\end{equation}
ここで $\rho(D_2) \approx 9.57/h^2$ は CCD $D_2$ 演算子の Fourier 解析による
スペクトル半径（付録~\ref{app:ccd_case2}；Eq-II 単独の理論上限 $9.6/h^2$
（式~\eqref{eq:lambda_max_theoretical}）の数値的近似）．
実測値（$N=32$: 3,038 回，$N=64$: 12,179 回，比 $\approx 4.0$）は $O(N^2)$ スケーリングを確認する．

\paragraph{CCD $+$ BiCGSTAB の収束．}
BiCGSTAB の収束ステップ数は CCD 係数行列の条件数
$\kappa(A_{\mathrm{CCD}}) = O(N^2)$ に依存するが，
最小残差型アルゴリズムにより実測値はほぼ $O(N)$（$N=64$: 37 回，$N=128$: 55 回）となった．
$N=256$ での 5 回という少数反復は，初期残差 $\|r_0\|$ が格子細分化により十分小さく
収束基準 $10^{-8}$ を少数ステップで達成したことによるものである．

\subsection{考察と結論}

\begin{enumerate}
  \item \textbf{精度}：CCD 系 3 手法はいずれも同一の CCD 演算子を用いるため，
        精度は同等（$N=256$ で $L_2 \approx 10^{-12}$）．
        FD2$+$SOR は $O(h^2)$ に制限され，$N=256$ 時点での精度差は約 4,000 倍．

  \item \textbf{反復回数}：FD2$+$SOR は $O(N)$（151 → 1151 for $N$: 32 → 256）；
        CCD$+$BiCGSTAB も概ね $O(N)$；
        CCD$+$仮想時間は $O(N^2)$（3,038 → 194,432 推定，64 倍増）．

  \item \textbf{メモリ}：仮想時間法は行列を保持せず $O(N)$ バイト（最小）；
        Direct LU と BiCGSTAB は係数行列を格納するため $O(N^2)$
        （1 次元では小さいが，2 次元で $N^2$ 倍，3 次元で $N^4$ 倍になる）．

  \item \textbf{推奨}：
        \begin{itemize}
          \item 小〜中規模 ($N \leq 128$，1 次元）: CCD$+$Direct LU が最速・最精度．
          \item 大規模・多次元（メモリ制約あり）：CCD$+$BiCGSTAB が精度と反復数のバランス最良．
          \item 行列構築が困難な非線形・移動境界問題：CCD$+$仮想時間が唯一の選択肢
                だが，$O(N^2)$ スケーリングが問題サイズを制限する（前処理が有効）．
        \end{itemize}
\end{enumerate}
