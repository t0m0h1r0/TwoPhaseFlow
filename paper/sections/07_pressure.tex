\section{圧力 Poisson 方程式（PPE）の導出と離散化}
\label{sec:pressure}

\subsection{変密度 Projection 法の段階的導出}
\label{sec:projection_derivation}

Projection 法（Chorin, 1968）の本質は，Navier-Stokes 方程式を
「\textbf{圧力なし予測}」と「\textbf{圧力による補正}」の2ステップに分解することにある．
変密度場ではこの分解に特別な注意が必要であり，以下に丁寧に導出する．

\subsubsection{Step 1：予測速度（Predictor）}

時刻 $t^{n+1}$ の完全な NS 方程式（無次元形）は：
\begin{equation}
  \rho^{n+1}\frac{\bu^{n+1} - \bu^n}{\Delta t}
  = \underbrace{-\rho^{n+1}(\bu^n\cdot\bnabla)\bu^n
    + \frac{1}{Re}\bnabla\cdot[\mu^{n+1}(\bnabla\bu^{n+1}+\bnabla^T\bu^{n+1})]
    + \bm{F}_\text{ext}^{n+1}}_{\displaystyle \equiv \bm{R}^{n+1}}
  - \bnabla p^{n+1}
  \label{eq:NS_full}
\end{equation}
圧力勾配項を除いた右辺 $\bm{R}^{n+1}$ から\textbf{予測速度} $\bu^*$ を定義する：
\begin{equation}
  \rho^{n+1}\frac{\bu^* - \bu^n}{\Delta t} = \bm{R}^{n+1}
  \quad\Longrightarrow\quad
  \bu^* = \bu^n + \frac{\Delta t}{\rho^{n+1}}\bm{R}^{n+1}
  \label{eq:predictor}
\end{equation}
$\bu^*$ は一般に発散フリーではない（$\bnabla\cdot\bu^* \neq 0$）．

\subsubsection{Step 2：圧力補正（Corrector）}

式 \eqref{eq:NS_full} から式 \eqref{eq:predictor} を引くと，
圧力と速度補正の関係が得られる：
\begin{equation}
  \rho^{n+1}\frac{\bu^{n+1} - \bu^*}{\Delta t} = -\bnabla p^{n+1}
  \quad\Longrightarrow\quad
  \bu^{n+1} = \bu^* - \frac{\Delta t}{\rho^{n+1}}\bnabla p^{n+1}
  \label{eq:corrector}
\end{equation}

\subsubsection{Step 3：発散フリー条件から PPE を導く}

$\bu^{n+1}$ に非圧縮条件 $\bnabla\cdot\bu^{n+1}=0$ を適用する．
式 \eqref{eq:corrector} の両辺に $\bnabla\cdot(\cdot)$ を作用させると：
\begin{align}
  \bnabla\cdot\bu^{n+1}
  &= \bnabla\cdot\bu^* - \Delta t\,\bnabla\cdot\!\left(\frac{1}{\rho^{n+1}}\bnabla p^{n+1}\right) = 0
  \notag \\
  &\Longrightarrow\quad
  \bnabla\cdot\!\left(\frac{1}{\rho^{n+1}}\bnabla p^{n+1}\right)
  = \frac{1}{\Delta t}\,\bnabla\cdot\bu^*
  \label{eq:PPE_derivation}
\end{align}

\begin{tcolorbox}[mybox, title={変密度 Projection 法の骨格（まとめ）}]
\begin{enumerate}
  \item \textbf{Predictor}：$\bu^* = \bu^n + \dfrac{\Delta t}{\rho^{n+1}}\bm{R}^{n+1}$
        \hfill（圧力なし，発散フリーでない速度）
  \item \textbf{PPE 求解}：$\bnabla\cdot\!\left(\dfrac{1}{\rho^{n+1}}\bnabla p^{n+1}\right) = \dfrac{1}{\Delta t}\bnabla\cdot\bu^*$
        \hfill（圧力を決定）
  \item \textbf{Corrector}：$\bu^{n+1} = \bu^* - \dfrac{\Delta t}{\rho^{n+1}}\bnabla p^{n+1}$
        \hfill（発散フリーに補正）
\end{enumerate}
\end{tcolorbox}

\begin{tcolorbox}[warnbox, title={なぜ $\bnabla\cdot(1/\rho\,\bnabla p)$ の形が出るのか？}]
一様密度（$\rho=\text{const}$）の場合，補正式 \eqref{eq:corrector} は
$\bu^{n+1} = \bu^* - (\Delta t/\rho)\bnabla p$ となり，発散を取ると
$\bnabla^2 p = \rho\,\bnabla\cdot\bu^*/\Delta t$ という標準的な Poisson 方程式になる．

変密度場では密度が空間的に変化するため，$\bnabla\cdot(1/\rho\,\bnabla p)$ という
\textbf{可変係数の拡散型演算子}が現れる（係数 $1/\rho$ が空間に依存）．
この演算子の離散化に調和平均密度を使う理由は6章で説明した通りである．
\end{tcolorbox}

\subsection{Projection 法と PPE}
\label{sec:ppe}

以上の導出から得られる変密度 Pressure Poisson Equation（PPE）を改めて記す：
\begin{equation}
  \bnabla\cdot\!\left( \frac{1}{\rho^{n+1}} \bnabla p^{n+1} \right) = \frac{1}{\Delta t}\, \bnabla\cdot \bu^{\ast}
  \label{eq:PPE_varrho}
\end{equation}

\subsection{FVM による離散化}

式\eqref{eq:PPE_varrho}は有限体積法（FVM）を用いて離散化する．
界面でのフラックス計算には，前節の Rhie--Chow 補正と整合するよう，
密度の調和平均を用いる．

\begin{tcolorbox}[mybox, title={FVM 変密度 PPE（Rhie-Chow 整合）}]
\begin{equation}
  \frac{1}{h_x^2}\!\left[
    \frac{p^{n+1}_{i+1,j}-p^{n+1}_{i,j}}{\tilde\rho_{i+1/2,j}}
  - \frac{p^{n+1}_{i,j}-p^{n+1}_{i-1,j}}{\tilde\rho_{i-1/2,j}}
  \right]
  + \cdots
  = \frac{(\bnabla_h^{\mathrm{RC}}\!\cdot\bu^*)_{i,j}}{\Delta t}
  \label{eq:PPE_FVM}
\end{equation}
\end{tcolorbox}
ここで $\tilde\rho_{i+1/2,j}$ は界面での調和平均密度である．
得られた連立一次方程式は BiCGSTAB 法などで求解する．

\subsection{境界条件}
\begin{itemize}
    \item \textbf{固体壁・流入出：} $\partial p/\partial n = 0$（ノイマン条件）
    \item \textbf{自由表面：} $p = p_{\mathrm{atm}}$（ディリクレ条件）
    \item \textbf{全周ノイマンの場合：} 解の一意性を保つため，1点の圧力を固定するか平均値を0にする拘束を加える．
\end{itemize}

\subsection{CCD 陰解法と仮想時間ステップ反復}
\label{sec:ppe_pseudotime}

\subsubsection{仮想時間ステップ法の定式化}

離散 PPE を
\begin{equation}
  \mathcal{L}_h\, p = q_h,
  \qquad
  q_h \equiv \frac{(\bnabla_h^{\mathrm{RC}}\cdot\bu^*)}{\Delta t}
  \label{eq:PPE_discrete}
\end{equation}
と書く（$\mathcal{L}_h$ は式 \eqref{eq:PPE_FVM} で与えられる可変係数 Laplace 演算子の離散化）．
この楕円型方程式を直接解く代わりに，\textbf{仮想時間} $\tau$ を導入した放物型問題
\begin{equation}
  \pd{p}{\tau} = q_h - \mathcal{L}_h\, p
  \label{eq:pseudo_parabolic}
\end{equation}
を考える．$\tau\to\infty$ で $\partial p/\partial\tau \to 0$ となり，式~\eqref{eq:pseudo_parabolic} の定常解は元の PPE~\eqref{eq:PPE_discrete} の解に一致する．

\subsubsection{仮想時間方向の陰的離散化}

式~\eqref{eq:pseudo_parabolic} を仮想時間方向に\textbf{陰的 Euler 法}で離散化する（$m$ は仮想時間反復ステップ）：
\begin{equation}
  \frac{p^{m+1} - p^m}{\Delta\tau} = q_h - \mathcal{L}_h\, p^{m+1}
  \label{eq:pseudo_implicit_raw}
\end{equation}
整理すると，各反復ステップで解くべき線形系が得られる：
\begin{equation}
  \bigl(\mathbf{I} + \Delta\tau\,\mathcal{L}_h\bigr)\, p^{m+1}
  = p^m + \Delta\tau\, q_h
  \label{eq:pseudo_linear}
\end{equation}

\begin{tcolorbox}[mybox, title={陰解法の安定性上の利点}]
式~\eqref{eq:pseudo_implicit_raw} の陰的離散化は，仮想時間刻み $\Delta\tau$ の大きさに関係なく\textbf{無条件安定}である．
これにより $\Delta\tau$ を任意に大きく設定でき，収束を加速することが可能である
（CFL 制約による $\Delta\tau$ の上限が存在しない）．
\end{tcolorbox}

\subsubsection{CCD 法による空間離散化}

式~\eqref{eq:pseudo_linear} の左辺行列 $\mathbf{I} + \Delta\tau\,\mathcal{L}_h$ を構成するにあたり，
空間演算子 $\mathcal{L}_h$ を\textbf{CCD 法}（5章）で離散化する．

2次元の PPE 演算子は
\begin{equation}
  \mathcal{L}_h = \mathcal{L}_x + \mathcal{L}_y,
  \qquad
  \mathcal{L}_x \equiv \pd{}{x}\!\Bigl(\tfrac{1}{\rho}\pd{p}{x}\Bigr),\quad
  \mathcal{L}_y \equiv \pd{}{y}\!\Bigl(\tfrac{1}{\rho}\pd{p}{y}\Bigr)
  \label{eq:L_split}
\end{equation}
と書ける．各方向の一次・二次微分を CCD で求め，$\mathcal{O}(h^6)$ の精度で $\mathcal{L}_h$ を評価する．
具体的には，各格子行（$i=$const）および格子列（$j=$const）に対して
式~\eqref{eq:ccd_global} のブロック Thomas 法を適用することで，
$\partial p/\partial x,\,\partial^2 p/\partial x^2$ および $\partial p/\partial y,\,\partial^2 p/\partial y^2$ を
$\mathcal{O}(N)$ の計算量で一括計算する．

\subsubsection{ADI 法との比較}

ADI 法（交互方向陰解法）では，演算子を $x$・$y$ 方向に交互に分割して解く：
\begin{equation}
  \Bigl(\mathbf{I} + \tfrac{\Delta\tau}{2}\mathcal{L}_x\Bigr)\tilde{p}
  = \Bigl(\mathbf{I} - \tfrac{\Delta\tau}{2}\mathcal{L}_y\Bigr)p^m + \Delta\tau\, q_h,
  \quad
  \Bigl(\mathbf{I} + \tfrac{\Delta\tau}{2}\mathcal{L}_y\Bigr)p^{m+1}
  = \Bigl(\mathbf{I} - \tfrac{\Delta\tau}{2}\mathcal{L}_x\Bigr)\tilde{p} + \Delta\tau\, q_h
  \label{eq:ADI}
\end{equation}
この分割は
$
  \bigl(\mathbf{I}+\tfrac{\Delta\tau}{2}\mathcal{L}_x\bigr)
  \bigl(\mathbf{I}+\tfrac{\Delta\tau}{2}\mathcal{L}_y\bigr)
  = \mathbf{I} + \Delta\tau\mathcal{L}_h + \tfrac{\Delta\tau^2}{4}\mathcal{L}_x\mathcal{L}_y
$
の関係から，$\mathcal{O}(\Delta\tau^2)$ の\textbf{分割誤差}（splitting error）を生じる．

本研究では ADI 分割を行わず，
仮想時間ステップ方向に対して直接，式~\eqref{eq:pseudo_linear} を各反復で繰り返す．
これにより，

\begin{itemize}
  \item 分割誤差がゼロ（各反復での圧力更新が全方向の残差を同時に低減）
  \item 反復回数を収束判定で制御でき，過剰計算を避けられる
  \item 複雑な密度場・非直交格子への拡張が容易
\end{itemize}

という利点が得られる（表~\ref{tab:ppe_methods} 参照）．

\subsubsection{収束判定と反復アルゴリズム}

収束判定には PPE の残差ノルムを用いる：
\begin{equation}
  \varepsilon^m \equiv \bigl\|\mathcal{L}_h p^m - q_h\bigr\|_2
  = \bigl\|(\mathbf{I}+\Delta\tau\mathcal{L}_h)p^m - (p^{m-1}+\Delta\tau q_h)\bigr\|_2 \big/ \Delta\tau
  \label{eq:residual}
\end{equation}
$\varepsilon^m < \varepsilon_{\rm tol}$（例：$\varepsilon_{\rm tol} = 10^{-8}$）を満たした時点で収束とみなし，反復を終了する．

\begin{tcolorbox}[mybox, title={PPE 仮想時間陰解法の反復アルゴリズム}]
\begin{enumerate}
  \item \textbf{初期値：} $p^0 = p^n$（前時刻の圧力）を初期推定値とする．
  \item \textbf{RHS 構築：} $q_h = (\bnabla_h^{\mathrm{RC}}\cdot\bu^*)/\Delta t$ を計算する．
  \item \textbf{反復ループ（$m = 0, 1, 2, \dots$）：}
    \begin{enumerate}
      \item CCD ブロック Thomas 法で各行・各列の空間微分を計算し，$\mathcal{L}_h p^m$ を評価．
      \item 残差 $\varepsilon^m = \|\mathcal{L}_h p^m - q_h\|_2$ を計算し，$\varepsilon^m < \varepsilon_{\rm tol}$ なら収束として終了．
      \item 線形系~\eqref{eq:pseudo_linear} を解いて $p^{m+1}$ を更新．
    \end{enumerate}
  \item \textbf{出力：} 収束した $p^{m+1} = p^{n+1}$ を圧力補正に渡す．
\end{enumerate}
\end{tcolorbox}
